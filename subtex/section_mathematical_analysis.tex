\section{Mathematical Analysis}

\begin{frame}{Limits and Continuity}
    \begin{itemize}
        \setlength{\itemsep}{0.5em}
        \setlength{\parskip}{0.2em}
        \item A function $f(x)$ is continuous at $x=a$ if
        \[
        \lim_{x \to a} f(x) = f(a)
        \]
        \item Continuity ensures small input changes produce small output changes.
        \item Typical discontinuities include removable, jump, and infinite discontinuities.
    \end{itemize}
\end{frame}

\begin{frame}{Derivative and Monotonicity}
    \begin{itemize}
        \setlength{\itemsep}{0.5em}
        \setlength{\parskip}{0.2em}
        \item The derivative is defined by
        \[
        f'(x)=\lim_{h\to 0}\frac{f(x+h)-f(x)}{h}
        \]
        \item If $f'(x) > 0$ on an interval, then $f(x)$ is increasing there.
        \item If $f'(x) < 0$ on an interval, then $f(x)$ is decreasing there.
        \item Critical points satisfy $f'(x)=0$ or $f'(x)$ does not exist.
    \end{itemize}
\end{frame}

\begin{frame}{Integral and Fundamental Theorem}
    \begin{itemize}
        \setlength{\itemsep}{0.5em}
        \setlength{\parskip}{0.2em}
        \item The definite integral measures signed area:
        \[
        \int_a^b f(x)\,dx
        \]
        \item If $F'(x)=f(x)$, then
        \[
        \int_a^b f(x)\,dx = F(b)-F(a)
        \]
        \item Differentiation and integration are linked by the fundamental theorem of calculus.
    \end{itemize}
\end{frame}
