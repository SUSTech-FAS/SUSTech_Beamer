\section{RWIP Lagrangian Modeling}

\begin{frame}{Model Setup and Assumptions}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \includegraphics[width=\linewidth]{lib/pic/动量轮立棍-有坐标系/动量轮立棍-无偏角-有坐标系.png}
        \end{column}
        \begin{column}{0.52\textwidth}
            \small
            \begin{itemize}
                \item Reaction wheel installation bias is ignored: $\alpha=0$
                \item Yaw is ignored: $\psi=0,\ \dot{\psi}=0$
                \item Attitude states: roll $\phi$, pitch $\theta$
                \item Platform + wheel mass: $m_p$
                \item COM in body frame: ${}^{\mathbf{b}}\!M_p=[0,0,l]^T$
                \item Gravity in world frame: ${}^{\mathbf{o}}\!g_p=[0,0,-m_pg]^T$
                \item Wheel relative rates: ${}^{\mathbf{b}}\!\omega_w=[p_{w1},p_{w2},0]^T$
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Kinematics and Inertia}
    \small
    \[
    {}^{\mathbf{o}}\!R_{\mathbf{b}}=R_y(\theta)R_x(\phi)=
    \begin{bmatrix}
    \cos\theta & \sin\theta\sin\phi & \sin\theta\cos\phi\\
    0 & \cos\phi & -\sin\phi\\
    -\sin\theta & \cos\theta\sin\phi & \cos\theta\cos\phi
    \end{bmatrix}
    \]
    \[
    {}^{\mathbf{b}}\!T_{\mathbf{o}}(\phi,\theta)=
    \begin{bmatrix}
    1 & 0 & -\sin\theta\\
    0 & \cos\phi & \sin\phi\cos\theta\\
    0 & -\sin\phi & \cos\phi\cos\theta
    \end{bmatrix},
    \quad
    {}^{\mathbf{o}}\!\dot E_p=\begin{bmatrix}\dot\phi\\\dot\theta\\0\end{bmatrix}
    \]
    \[
    {}^{\mathbf{b}}\!\omega_p=
    {}^{\mathbf{b}}\!T_{\mathbf{o}}\,{}^{\mathbf{o}}\!\dot E_p=
    \begin{bmatrix}\dot\phi\\\dot\theta\cos\phi\\-\dot\theta\sin\phi\end{bmatrix}
    \]
    \[
    {}^{\mathbf{b}}_{\mathbf{o}}\!I_p=\mathrm{diag}(J_{p1},J_{p2},J_{p3}),\quad
    {}^{\mathbf{b}}\!I_w=\mathrm{diag}(I_{w1},I_{w1},0)
    \]
\end{frame}

\begin{frame}{Lagrangian}
    \small
    \[
    K_p=\frac{1}{2}\,{}^{\mathbf{b}}\!\omega_p^T\,{}^{\mathbf{b}}_{\mathbf{o}}\!I_p\,{}^{\mathbf{b}}\!\omega_p
    \]
    \[
    {}^{\mathbf{b}}\!\omega_{w,\text{total}}=
    {}^{\mathbf{b}}\!\omega_p+{}^{\mathbf{b}}\!\omega_w=
    \begin{bmatrix}
    \dot\phi+p_{w1}\\
    \dot\theta\cos\phi+p_{w2}\\
    -\dot\theta\sin\phi
    \end{bmatrix}
    \]
    \[
    K_w=\frac{1}{2}\,{}^{\mathbf{b}}\!\omega_{w,\text{total}}^T
    {}^{\mathbf{b}}\!I_w\,{}^{\mathbf{b}}\!\omega_{w,\text{total}}
    \]
    \[
    P={}^\mathbf{o}\!g_p^T\,{}^\mathbf{o}\!R_\mathbf{b}\,{}^\mathbf{b}\!M_p
    =-m_pgl\cos\phi\cos\theta
    \]
    \[
    L=K_p+K_w-P
    \]
\end{frame}

\begin{frame}{Euler-Lagrange Equations}
    \small
    \[
    \frac{d}{dt}\left(\frac{\partial L}{\partial \dot q_i}\right)-\frac{\partial L}{\partial q_i}=Q_i
    \]
    \begin{itemize}
        \item Platform coordinates:
        \[
        \frac{d}{dt}\left(\frac{\partial L}{\partial \dot\phi}\right)-\frac{\partial L}{\partial \phi}=0,\quad
        \frac{d}{dt}\left(\frac{\partial L}{\partial \dot\theta}\right)-\frac{\partial L}{\partial \theta}=0
        \]
        \item Wheel coordinates:
        \[
        \frac{d}{dt}\left(\frac{\partial L}{\partial p_{w1}}\right)=\tau_1,\quad
        \frac{d}{dt}\left(\frac{\partial L}{\partial p_{w2}}\right)=\tau_2
        \]
        \item Gravity terms are already included in $L$ via $P$.
    \end{itemize}
\end{frame}

\begin{frame}{Final Dynamics I (Platform, No Yaw)}
    \footnotesize
    \[
    \begin{aligned}
    &\frac{I_{w1}\dot{\theta}^{2}\sin(2\phi)}{2}
    +I_{w1}\dot{\theta}p_{w2}\sin\phi
    +I_{w1}\dot p_{w1}\\
    &+\frac{J_{p2}\dot{\theta}^{2}\sin(2\phi)}{2}
    -\frac{J_{p3}\dot{\theta}^{2}\sin(2\phi)}{2}
    +(I_{w1}+J_{p1})\ddot\phi\\
    &+glm_p\sin\phi\cos\theta=0
    \end{aligned}
    \]
    \[
    \begin{aligned}
    &I_{w1}\dot p_{w2}\cos\phi
    +\ddot\theta\!\left(I_{w1}\cos^2\phi+J_{p2}\cos^2\phi+J_{p3}\sin^2\phi\right)\\
    &-\dot\phi\!\left(2I_{w1}\dot\theta\cos\phi+I_{w1}p_{w2}\right)\sin\phi\\
    &-\dot\phi\!\left(2J_{p2}\dot\theta\cos\phi-2J_{p3}\dot\theta\cos\phi\right)\sin\phi\\
    &+glm_p\sin\theta\cos\phi=0
    \end{aligned}
    \]
\end{frame}

\begin{frame}{Final Dynamics II (Wheel, No Yaw)}
    \footnotesize
    \[
    I_{w1}\left(\ddot\phi+\dot p_{w1}\right)=\tau_1
    \]
    \[
    \begin{aligned}
    I_{w1}\bigl(\ddot\theta\cos\phi-\dot\phi\dot\theta\sin\phi+\dot p_{w2}\bigr)=\tau_2
    \end{aligned}
    \]
\end{frame}
